OJS Clickable d3.Treemap

This is a clickable treemap using Observable JS to render a D3 hierarchy. It features centered labels, a hover effect, a custom HTML floating tooltip and a Clickable target URL.

data = ({
  name: "root",
  children: [
    { 
      name: "Frontend", 
      children: [
        {name: "React", value: 100, url: "https://react.dev"}, 
        {name: "Vue", value: 60, url: "https://vuejs.org"}, 
        {name: "Svelte", value: 40, url: "https://svelte.dev"}
      ] 
    },
    { 
      name: "Backend", 
      children: [
        {name: "Node.js", value: 80, url: "https://nodejs.org"}, 
        {name: "Python", value: 90, url: "https://www.python.org"}, 
        {name: "Go", value: 50, url: "https://go.dev"}
      ] 
    },
    { 
      name: "DevOps", 
      children: [
        {name: "Docker", value: 70, url: "https://www.docker.com"}, 
        {name: "K8s", value: 110, url: "https://kubernetes.io"}
      ] 
    }
  ]
})

width = 800
height = 500
fontSize = "24px" // <--- ADJUST FONT SIZE HERE

// --- 2. HIERARCHY CALCULATION ---
root = {
  const h = d3.hierarchy(data)
      .sum(d => d.value)
      .sort((a, b) => b.value - a.value);
  
  return d3.treemap()
      .size([width, height])
      .paddingInner(2)
      (h);
}

// --- 3. TOOLTIP ELEMENT ---
tooltip = d3.select("body").append("div")
    .style("position", "absolute")
    .style("visibility", "hidden")
    .style("background", "rgba(0, 0, 0, 0.8)")
    .style("color", "#fff")
    .style("padding", "8px")
    .style("border-radius", "4px")
    .style("font-size", "12px")
    .style("pointer-events", "none")
    .style("z-index", "1000")
    .style("font-family", "sans-serif");

// --- 4. CHART RENDERING ---
chart = {
  const svg = d3.create("svg")
      .attr("viewBox", [0, 0, width, height])
      .attr("width", "100%")
      .attr("height", "auto")
      .style("font", "12px sans-serif");

  // Create groups for each leaf
  const leaf = svg.selectAll("g")
    .data(root.leaves())
    .join("g")
    .attr("transform", d => `translate(${d.x0},${d.y0})`);

  // Wrap each node in an anchor (<a>) tag for URL navigation
  const link = leaf.append("a")
    .attr("xlink:href", d => d.data.url)
    .attr("target", "_blank")
    .style("cursor", "pointer");

  // Draw Rectangles inside the anchor tag
  link.append("rect")
    .attr("fill", d => { 
        let p = d; while (p.depth > 1) p = p.parent; 
        return d3.schemeTableau10[data.children.indexOf(p.data) % 10]; 
    })
    .attr("fill-opacity", 0.7)
    .attr("width", d => d.x1 - d.x0)
    .attr("height", d => d.y1 - d.y0)
    .attr("stroke", "white")
    .on("mouseover", function(event, d) {
      d3.select(this).attr("fill-opacity", 1);
      tooltip.style("visibility", "visible")
             .html(`<strong>${d.data.name}</strong><br/>Value: ${d.value}`);
    })
    .on("mousemove", function(event) {
      tooltip.style("top", (event.pageY - 10) + "px")
             .style("left", (event.pageX + 10) + "px");
    })
    .on("mouseout", function() {
      d3.select(this).attr("fill-opacity", 0.7);
      tooltip.style("visibility", "hidden");
    });

  // Draw Centered Labels inside the anchor tag
  link.append("text")
    .attr("x", d => (d.x1 - d.x0) / 2)
    .attr("y", d => (d.y1 - d.y0) / 2)
    .attr("dy", "0.35em")
    .attr("text-anchor", "middle")
    .attr("fill", "white")
    .style("font-size", fontSize)
    .style("font-weight", "bold")
    .style("pointer-events", "none") // Hover passes through text to the rect
    .text(d => (d.x1 - d.x0 > 40 && d.y1 - d.y0 > 20) ? d.data.name : "");

  return svg.node();
}
